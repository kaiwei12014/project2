def load_snp_close():
f = open('table.csv', 'rb').readlines()[1:]
raw_data = []
raw_dates = []
for line in f:
try:
close_price = float(line.split(',')[4])
raw_data.append(close_price)
raw_dates.append(line.split(',')[0])
except:
continue
return raw_data, raw_dates
